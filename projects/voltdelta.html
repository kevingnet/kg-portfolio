void CAdminSocket::OnStateChanged( SocketState State )
{
    OWS_1trace( (*m_pCTrace), LEVEL_EVENTSANDPOLLS, "Socket state changed to: %s", GetStateString() );
	switch( State ) 
	{
		case IP_STATE_CLOSED:
			OWS_0trace( (*m_pCTrace), LEVEL_INFO, "Network disconnected, enabling listen port" );
			m_pListenAdminSocket->OpenListenPort( CAdminSocket::sm_iBasePort );
			break;
		case IP_STATE_CONNECTED:
			OWS_0trace( (*m_pCTrace), LEVEL_INFO, "Network connected, disabling listen port" );
			m_pListenAdminSocket->ShutDown();
			m_pListenAdminSocket->Close();
			break;
		case IP_STATE_CREATED:
		case IP_STATE_CONNECTING:
		case IP_STATE_LISTENING:
		case IP_STATE_DISCONNECTING:
		default:
			break;
	}
}

void CAdminSocket::OnReceive( int /*nErrorCode*/ )
{
	int		iLastError = 0;
	sm_Count = Receive( sm_Buffer, sizeof(sm_Buffer) );
	switch( sm_Count )
	{   
		case 0: //no data?, oh! well...
			OWS_0trace( (*m_pCTrace), LEVEL_DATATRANSFERS, "Received 0 bytes, ???" );
			return;
			break;   
		case SOCKET_ERROR:
			iLastError = GetLastError();
			if( iLastError != WSAEWOULDBLOCK )       
			{
				OWS_0trace( (*m_pCTrace), LEVEL_ERROR, "Receive error" );
				DisplayLastError( iLastError );
			}
			else
			{
				OWS_0trace( (*m_pCTrace), LEVEL_WARNING, "Could not receive data, socket is currently blocking, will retry later." );
			}
			return;
			break;
		default:      
			OWS_1trace( (*m_pCTrace), LEVEL_DATATRANSFERS, "Received %s bytes", sm_Count );
			break;
	}

	int iPVCNumber = sm_Buffer[0];
	iPVCNumber <<= 8;
	iPVCNumber += sm_Buffer[1];

	LPX25PVC pPVC = m_pX25Manager->GetPVC( iPVCNumber );

	switch( sm_Buffer[2] )
	{
		case GET_LINE_STATE:
			OWS_1trace( (*m_pCTrace), LEVEL_OPERATIONS, "Remote requested line %i state", iPVCNumber );
			SendReplyMessage( iPVCNumber, GET_LINE_STATE, pPVC->GetPVCState() );
			break;
		case SET_LINE_STATE:
			OWS_1trace( (*m_pCTrace), LEVEL_OPERATIONS, "Remote requested line %i state change, cannot comply with request", iPVCNumber );
			//SendReplyMessage( iPVCNumber, GET_LINE_STATE, GetPVCStateString() );
			break;
		case GET_LINE_STATISTICS:
		{
			OWS_1trace( (*m_pCTrace), LEVEL_OPERATIONS, "Remote requested line %i statistics", iPVCNumber );

			CString szString;

			szString = "LINE STAT\nPhysical Layer:\n";
			szString += pPVC->GetStatistics( CIDTYPE_PHYSICAL );
			szString += "\nLink Layer:\n";
			szString += pPVC->GetStatistics( CIDTYPE_LINK );
			szString += "\nPacket Layer:\n";
			szString += pPVC->GetStatistics( CIDTYPE_PACKET );
			szString += "\nPVC:\n";
			szString += pPVC->GetStatistics( CIDTYPE_PVC );

			SendReplyMessage( iPVCNumber, GET_LINE_STATISTICS, pPVC->GetPVCState(), szString );
		}
			break;
		case CLEAR_LINE_STATISTICS:
			pPVC->ClearStatistics();
			break;
		default:
			OWS_1trace( (*m_pCTrace), LEVEL_WARNING, "Unknown request for line %i", iPVCNumber );
			break;
	}
}

void CAdminSocket::SendReplyMessage( int iLine, int iMessageType, int iCode, LPCTSTR szDetails )
{
	sm_Buffer[0]	= MESSAGE_TYPE_REPLY;
	sm_Buffer[1] = HIBYTE(iLine);
	sm_Buffer[2] = LOBYTE(iLine);
	sm_Buffer[3]	= iMessageType;
	sm_Buffer[4]	= iCode;
	sm_Count		= 5;
	if( szDetails && *szDetails )
	{
		sm_Count += strlen( szDetails );
		strcpy( sm_Buffer+5, szDetails );
	}
	Send( (PBYTE)sm_Buffer, sm_Count );
}

Procedure to test the X.25 boards in two computers in a back to back configuration Locate PCI Slot 1 in each computer. These boards like PCI slot 1 and would not work correctly otherwise, they just would seem to work. Refer to the vendor documentation and setup one board for DCE operation. The other board should be set for DTE operation, but check anyway. Setup the NT driver in each computer according to the directions. Before continuing make sure that you start the device driver from the control panel on both computers. Connect the octopus cable to each board and locate line one on each cable. Prepare a null modem cable or RS-232 break-out box according to the following configuration: 1 ►1 (this is already jumpered by default) 2 ►3 3 ►2 4 ►5 5 ►4 7 ►7 15 ►17 17 ►15 6 & 8 ►20 20 ►6 & 8 Connect both lines 1 from each octopus cable to the break-out box. I have already setup the configuration (*.cfg) files for the tests. I also setup batch files DCE.bat and DTE.bat. Open a command prompt window and run DTE.bat in the DTE machine and DCE.bat in the DCE machine. You will see a GUI test application called WinTDrv. The documentation for that app is in our vendor provided HTML documents that you should be familiar with. Click on the Open button on both computers. Click on the Listen button on one, click on connect on the other one. You should know see that they are in Data mode Xon. Click on the Full Duplex button on both of them. You should see that they are both transmitting and receiving data. That concludes the back to back test. Now we can get on with the ftp test so that we can be confident that the two computers can do something useful. We will transfer a few files from one to the other via X.25. I’ve setup the DCE as the master (will transmit files) and the DTE as slave (will receive files). If you want to change that you must edit slave.cfg and master.cfg according to ARC’s documentation and change the clock mode setting, that’s all you should need to change. Read sftptest.txt and follow those directions. In the same command prompt window run Slave.bat on the DTE computer and Master.bat in the DCE computer. In a few seconds you should see that they are transferring files from the DCE to the DTE. When finished make sure that the files were actually copied to the DTE. NOTE: This configuration took a few days to complete. If you want to save yourself time read our vendor documentation so that you familiarize yourself with the hardware and software and then follow these directions in this file carefully. Most of the configuration work has been done and it is contained in the *.cfg files. Those files have not yet been optimized, so you may change parameters in them to optimize the performance for the platform and type of application they will be part of.

void CX25Line::ConnectLayer( CIDType iCIDType )
{
	CID cid = lgo_INVALID_CID;
	switch( iCIDType )
	{
		case CIDTYPE_PACKET:
			cid = m_CIDPacket;
			break;
		case CIDTYPE_LINK:
			cid = m_CIDLink;
			break;
		case CIDTYPE_PHYSICAL:
			cid = m_CIDPhysical;
			break;
	}
	m_Result = 0;
	if( STATE_OPEN == lgo_State( cid ) )
	{
		m_Result = lgo_ConnectRequest( cid, NULL, 0 );
		ReportError( "Connect Request", m_Result, iCIDType, true );
	}
}

void CX25Line::DisconnectLayer( CIDType iCIDType )
{
	CID cid = lgo_INVALID_CID;
	switch( iCIDType )
	{
		case CIDTYPE_PACKET:
			cid = m_CIDPacket;
			break;
		case CIDTYPE_LINK:
			cid = m_CIDLink;
			break;
		case CIDTYPE_PHYSICAL:
			cid = m_CIDPhysical;
			break;
	}
	m_Result = 0;
	m_StateCode = lgo_State( cid );
	if( STATE_DATA_TRANSFER_XON == m_StateCode ||
		STATE_RESETTING == m_StateCode )
	{
		m_Result = lgo_DisconnectRequest( cid, NULL, 0 );
		ReportError( "Disconnect Request", m_Result, iCIDType, true );
		m_Result = lgo_Reopen( cid );
		ReportError( "Reopen", m_Result, iCIDType );
	}
}

void CX25Line::ListenLayer( CIDType iCIDType )
{
	CID cid = lgo_INVALID_CID;
	switch( iCIDType )
	{
		case CIDTYPE_PACKET:
			cid = m_CIDPacket;
			break;
		case CIDTYPE_LINK:
			cid = m_CIDLink;
			break;
		case CIDTYPE_PHYSICAL:
			cid = m_CIDPhysical;
			break;
	}
	m_StateCode = lgo_State( cid );
	switch( m_StateCode )
	{
		case STATE_LISTENING:
			m_Result = 0;
			break;
		case STATE_OPEN:
			m_Result = lgo_Listen( cid );
			ReportError( "Listen", m_Result, iCIDType );
			break;

		case STATE_WAITING_FOR_REMOTE_ACCEPT:
		case STATE_WAITING_FOR_REMOTE_CONFIRMATION:

		case STATE_WAITING_FOR_LOCAL_CONFIRMATION:
		case STATE_WAITING_FOR_LOCAL_ACCEPT:

		case STATE_RESETTING:
		case STATE_DATA_TRANSFER_XOFF:
		case STATE_DATA_TRANSFER_XON:
			m_Result = lgo_Reopen( cid );
			ReportError( "Reopen", m_Result, iCIDType );
			m_Result = lgo_Listen( cid );
			ReportError( "Listen", m_Result, iCIDType );
			break;
	}
}

bool CX25Line::AllLayersOpen()
{
	if( STATE_OPEN == lgo_State( m_CIDPacket ) &&
		STATE_OPEN == lgo_State( m_CIDLink ) &&
		STATE_OPEN == lgo_State( m_CIDPhysical ) )
	{
		return true;
	}
	return false;
}

{
	if( STATE_DATA_TRANSFER_XON == lgo_State( m_CIDPacket ) &&
		STATE_DATA_TRANSFER_XON == lgo_State( m_CIDLink ) &&
		STATE_DATA_TRANSFER_XON == lgo_State( m_CIDPhysical ) )

	return false;
}

{
	ListenLayer( CIDTYPE_PHYSICAL );
	if( m_Result >= 0 )

		ListenLayer( CIDTYPE_LINK );

				ChangeState( LINE_STATE_LISTENING );
			}
		}
	}

Overview 2.1 General X.25 Module is actually an application that acts as a gateway between TCP/IP and the X.25 Protocols. X.25 is a widely used protocol in legacy systems. The telecommunications industry relies on this protocol. However many companies are moving from X.25 to TCP/IP, many speculate this protocol would soon become obsolete. Nevertheless, because a great number of systems use this protocol, we must support it. X.25 Module will address two main issues: a) We must support X.25 and b) We must also move on to newer technologies. Therefore, the purpose of this module is to provide X.25 connectivity to any client in such a way that the client application ‘thinks’ it’s a TCP/IP connection. Thus we can support X.25 while allowing other programmers to design their applications for TCP/IP connectivity. This method offers other advantages, the most obvious one being easy migration to TCP/IP. The X.25 Module is an MFC Dialog Box Application. It has a GUI with menus and other controls which provide limited functionality (functionality is sort of automatic and services are provided to TCP/IP clients.) The module uses Emulex xp8400 hardware which provides X.25 support. The drivers are provided by ARC (Advanced Relay Corp.) The network connectivity to X.25 (including physical, link, packet and PVC layers) is provided in an API in DLL’s also by ARC. Configuration information for the stack, system buffers, physical, link and packet layers is stored in several configuration files (*.cfg) in a location pointed to by the configuration database (MS Access database.) This configuration files contain parameters specific to X.25 such as DTE or DCE operation, number of PVCs, packet sizes, etc…, for complete information refer to ARC’s documentation. Each X.25 Line has a number of PVCs. For each PVC there is a TCP/IP listen socket, which remains listening until a connection is made for efficient resource utilization. Application settings and configuration information is stored in an MS Access database. Access to this database is through ADO via a ODBC DSN. Connectivity to the database is stored in the registry. The database stores: 1. Defaults 2. Connection types Component Design Document X.25 Module 3. Protocol names, and 4. Services. Each entry is an X.25 line and describes which *.cfg files to apply to it, how many PVCs it will use and its connection type [listen, connect, disabled] The user interface provides icons and simulated LED’s that reflect the current TCP/IP socket, X.25 Line and PVC states. An admin listen socket is provided so that applications can control the module and request information for specific lines. Care must be taken when configuring the lines. It has been known to ARC that the hardware has become damaged from incorrect settings. Also the number of PVCs must match per line. Trace messages are generated for just about anything, except in places where this would cause the module to slow down. For information on Trace, refer to its document. 2.2 Context Three possible contexts:  User  TCP/IP Client  TCP/IP Client Admin 2.3 Interfaces USER INTERFACE: Menus:  Startup. Initialize lines  Shutdown. De-initialize lines  Configuration Database. Call ADO dialog box to set connectivity parameters  Exit.  Statistics. The listen socket ports are numbered after the Base Port parameter in the Defaults table in the configuration database. For example if you use base port 2000 that port will be the Admin port, and PVC #1 will be available on port 2001, PVC #2 on port 2002, etc…, To send data to the PVC simply send the data on the connected port. Because currently the ARC API and the xp8400 driver do not support call back events, so we do not know when the data arrives, the module must poll for data events as fast as needed. That isn’t necessarily a problem, though. TCP can process data with higher latency than X.25 and the new PC’s CPUs are a magnitude of times faster than the data transmitted in either protocol anyway. So, when the module polls and finds data, sends it immediately through the client socket. This would give the client application the appearance that is dealing with a TCP/IP connection, indeed it is. The TCP/IP connectivity is provided through a DLL that encapsulates CAsyncSocket and provides additional functionality:  Network error reporting  Buffering of data  State transitions and notifications of them If the socket is trying to send data faster than X.25 can transmit, hopefully the other end will notify us. If so, an Xoff request is made by the other end. A message is then sent to the client indicating a blocking condition. The client should handle this, but can choose to ignore. The data is then buffered up to a point (default table defined). When the buffer is about to fill up and data keeps arriving warnings are sent to the client. If the client ignores those warnings and we still have not processed the data, the data is then discarded and an error message is issued back to the client. The protocol messages between the client and the module are conducted through the same TCP/IP connection, however it used the Out of Band Data mechanism. Clients connecting to the module should handle OOB data. They could ignore the data, at the risk of having the module discard data should the need arise. And they could not issue commands or request info from the module, of course. If the client socket disconnects or the connection is lost the PVC will again enable the listen socket. The message format is the same as that for the Admin client. With the following two differences: Component Design Document X.25 Module 1. Line Number is not used. The two bytes are NOT included [they are NOT zero] 2. Commands: Startup and Shutdown are not supported The module was coded so that the X.25 PVC would be resilient and provide fail-safe operation. Thus you can disconnect the cables from the xp8400 card and reconnect them, and the module will attempt to re-establish the connections. This depends on other factors, however. If the line was configured to listen, then, the other end is responsible for reconnecting. This is not the same listen as the listening socket, though. CONSIDERATIONS:  Do not send TCP/IP packets that are bigger than the X.25 max packet size setting in the *.cfg file for that line. Doing so would cause an error.  Lines that will not be used should be marked as disabled. The system uses resources and memory buffers for each active line, affecting performance. Routine Breakdown 4.1 Introduction The TCP/IP DLL encapsulates all protocol connectivity and provides automatic buffering. It uses CAsyncSocket. The LayGo DLLs encapsulate X.25 setup, initialization, buffers creation and maintenance, state transitions, configuration and all X.25 connectivity issues. The X.25 module is an MFC dialog box application. It has a timer that polls for data and events. It tries to ensure that we have a connection at all times and buffers data when the peer or the module are blocking and sends protocol messages as well as commands. 4.2 Module Hierarchy See graphic above. 4.3 Module Descriptions LayGo API, for a description and details reffer to ARC’s documentation The TCP/IP DLL is an FSM (finite state machine). It is used as an CAsyncSocket but provides automatic buffering and notification of state changes The X.25 module is structured as follows:  CX25RelayApp, (CWinApp) Initializes and deinitializes COM and WinSock2  CX25RelayDlg, (CDialog) provides GUI and contains X.25 manager object and CCNTrace object. Contains the timer function  CX25Manager, starts up and shuts down, accesses configuration database and configuration files, sets up connectivity, contains X.25 line and PVC objects, calls event and data handling functions in each line and PVC object, contains admin socket which accepts commands  CX25Line, is an FSM, contains network, link and packet layers and buffer used by its children PVCs, tries to ensure we have a good medium through which its PVCs can send and receive data, so it handles connections and network events  CX25PVC, derived from CTCPSocket is an FSM, contains minor device (PVC), contains buffers used when peer or local are blocking, sends and receives and acts on protocol messages, exchanges data, basically most of the action happens here Appendix

	//in this case because line is down we can work on the local
	ChangeState( LINE_TYPE_LOCAL_PVC, LINE_STATE_DOWN );

	//so we don't get data from peer socket...

	m_baSocketBuffer.RemoveAll();
