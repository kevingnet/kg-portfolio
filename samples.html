Word Finder — C++

Longest word buildable by concatenating shorter dictionary words.

bool can_build(const string& word, const set<string>& dict) {
  vector<bool> ok(word.size() + 1);
  ok[0] = true;
  for (size_t i = 1; i <= word.size(); i++)
    for (size_t j = 0; j < i; j++)
      if (ok[j] && dict.count(word.substr(j, i - j))) { ok[i] = true; break; }
  return ok[word.size()];
}
